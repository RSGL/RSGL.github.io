93 struct point{
94 bool operator==(point p){ return x == p.x && y == p.y; }
95 void operator+=(point p){ x += p.x; y += p.y; }
96 void operator-=(point p){ x -= p.x; y -= p.y; }
97 int x , y ;
98 };
99 struct area{
100 bool operator==(area a){ return w == a.w && h == a.h; }
101 void operator+=(area a){ w += a.w; h += a.h; }
102 void operator-=(area a){ w -= a.w; h -= a.h; }
103 int w , h ;
104 };
